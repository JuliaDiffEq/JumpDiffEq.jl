94 # adapted verion of Random.SamplerRangeFast for native 64 bits generation
95 # we don't use "near division-less algorithm", as it doesn't work for Int128,
96 # and we want to maintain as little code as possible here

98 using Base: BitUnsigned, BitInteger
99 using Random: LessThan, Masked, uniform

101 struct SamplerRangeFast{U<:BitUnsigned,T<:BitInteger} <: Sampler{T}
102 a::T # first element of the range
103 bw::UInt # bit width
104 m::U # range length - 1
105 mask::U # mask generated values before threshold rejection
106 end
107
108 SamplerRangeFast(r::AbstractUnitRange{T}) where T<:BitInteger =
109 SamplerRangeFast(r, T <: Base.BitInteger64 ? UInt64 : UInt128)
110
111 function SamplerRangeFast(r::AbstractUnitRange{T}, ::Type{U}) where {T,U}
112 isempty(r) && throw(ArgumentError("range must be non-empty"))
113 m = (last(r) - first(r)) % unsigned(T) % U
114 # ^--- % unsigned(T) to not propagate sign bit
115 bw = (sizeof(U) << 3 - leading_zeros(m)) % UInt # bit-width
116 mask = ((1 % U) << bw) - (1 % U)
117 SamplerRangeFast{U,T}(first(r), bw, m, mask)
118 end

120 function rand(rng::LehmerRNG, sp::SamplerRangeFast{UInt64,T}) where T
121 a, bw, m, mask = sp.a, sp.bw, sp.m, sp.mask
122 x = rand(rng, LessThan(m, Masked(mask, uniform(UInt64))))
123 (x + a % UInt64) % T
124 end

126 function rand(rng::LehmerRNG, sp::SamplerRangeFast{UInt128,T}) where T
127 a, bw, m, mask = sp.a, sp.bw, sp.m, sp.mask
128 x = bw <= 64 ?

130 LessThan(m % UInt64,

132 rand(rng, LessThan(m, Masked(mask, uniform(UInt128))))

147 function _shuffle!(r::StableRNG, a::AbstractArray)
148 require_one_based_indexing(a)
149 n = length(a)
150 n <= 1 && return a # nextpow below won't work with n == 0
151 @assert n <= Int64(2)^52
152 mask = nextpow(2, n) - 1
153 for i = n:-1:2
154 (mask >> 1) == i && (mask >>= 1)
155 j = 1 + rand(r, Random.ltm52(i, mask))
156 a[i], a[j] = a[j], a[i]
157 end
158 return a
159 end

162 @noinline function Random.randn_unlikely(rng::StableRNG, idx, rabs, x)
163 @inbounds if idx == 0
164 while true
165 xx = -Random.ziggurat_nor_inv_r*log(rand(rng))
166 yy = -log(rand(rng))
167 yy+yy > xx*xx &&
168 return (rabs >> 8) % Bool ? -Random.ziggurat_nor_r-xx : Random.ziggurat_nor_r+xx
169 end
170 elseif (Random.fi[idx] - Random.fi[idx+1])*rand(rng) + Random.fi[idx+1] < exp(-0.5*x*x)
171 return x # return from the triangular area
172 else
173 return randn(rng)
174 end
175 end

112 # dependency graph is a map from a reaction to a vector of reactions
113 # that should depend on species it changes
114 function make_dependency_graph(numspec, ma_jumps::AbstractMassActionJump)
115 numrxs = get_num_majumps(ma_jumps)
116 spec_to_dep_rxs = var_to_jumps_map(numspec, ma_jumps)
117
118 # create map from rx to reactions depending on it
119 dep_graph = [Vector{Int}() for n in 1:numrxs]
120 for rx in 1:numrxs
121
122 # rx changes spec, hence rxs depending on spec depend on rx
123 for (spec, stoch) in ma_jumps.net_stoch[rx]
124 for dependent_rx in spec_to_dep_rxs[spec]
125 push!(dep_graph[rx], dependent_rx)
126 end
